Material Progress

Type1: Determinate

var mprogress = new Mprogress();

start end set inc

mprogress.start() — Start and show progress bar.

mprogress.end() — Finish and hide bar.

mprogress.set(0.4) — Sets a percentage.

mprogress.inc() — Increase by a little.

Type2: Buffer

var mprogress2 = new Mprogress({ template: 2, parent: '#demoBuffer' });

start end

Type3: Indeterminate

var mprogress3 = new Mprogress({ template: 3, parent: '#demoIn' });

start end

var mprogress4 = new Mprogress({ template: 4, parent: '#demoQuery' });

start end
